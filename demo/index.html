fadeup

A simple, jQuery plugin for fancy hiding.

/* * * * * * * * * * * * * * * * *
 * On click of demo button,
 * fade up the content.
 * * * * * * * * * * * * * * * * */
$('.demo-1').on('click', function() {
  $('.demo-1-to-fade').fadeUp({
    duration: 400,
    easing: 'swing'
  });
});
